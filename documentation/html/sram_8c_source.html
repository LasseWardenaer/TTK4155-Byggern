32  uint16_t ext_ram_size = 0x800;
33  uint16_t write_errors = 0;

35  printf("Starting SRAM test...\n");
36  // rand() stores some internal state, so calling this function in a loop will
37  // yield different seeds each time (unless srand() is called before this function)
38  uint16_t seed = rand();
39  // Write phase: Immediately check that the correct value was stored
40  srand(seed);
41  for (uint16_t i = 0; i < ext_ram_size; i++) {
42  uint8_t some_value = rand();
43  ext_ram[i] = some_value;
44  uint8_t retreived_value = ext_ram[i];
45  if (retreived_value != some_value) {
46  printf("Write phase error: ext_ram[%4d] = %02X (should be %02X)\n", i, retreived_value, some_value);
47  write_errors++;
48  }
49  }
50  // Retrieval phase: Check that no values were changed during or after the write phase
51  srand(seed);
52  // reset the PRNG to the state it had before the write phase
53  for (uint16_t i = 0; i < ext_ram_size; i++) {
54  uint8_t some_value = rand();
55  uint8_t retreived_value = ext_ram[i];
56  if (retreived_value != some_value) {
57  printf("Retrieval phase error: ext_ram[%4d] = %02X (should be %02X)\n", i, retreived_value, some_value);
58  retrieval_errors++;
59  }
60  }
61  printf("SRAM test completed with \n%4d errors in write phase and \n%4d errors in retrieval phase\n\r", write_errors, retrieval_errors);
