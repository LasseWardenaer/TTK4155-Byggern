54  // enable clock for TC0: DIV = 0 (clk = MCK), CMD = 0 (read), PID = 27 (TC0)
55  PMC->PMC_PCR = PMC_PCR_EN | PMC_PCR_DIV_PERIPH_DIV_MCK | (ID_TC0 << PMC_PCR_PID_Pos);
56  PMC->PMC_PCER0 |= 1 << (ID_TC0);
57 
58  // disable timer counter channel
59  TC0->TC_CHANNEL[0].TC_CCR = TC_CCR_CLKDIS;
60 
61  // set clock to MCK/2 = 42 MHz, capture mode with reset trigger on compare match with RC

63 
64  // set match frequency to 50 Hz
