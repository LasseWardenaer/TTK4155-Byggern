303 void menu_run() {
304  char joystick_button = user_input_buttons().joystick;
305  if (joystick_button && !joystick_pushed) {
306  joystick_pushed = 1;
307 
308  if (current->action_functions[state]) {
309  current->action_functions[state]();
310  state_changed = 1;
311  }
312  }
313  else if (!joystick_button) {
314  joystick_pushed = 0;
315  }
316 
317  switch (user_input_joystick_dir()) {
318  case UP:
319  if (!joystick_centered) {
320  break;
321  }
322 
323  if (state == 0) {
324  state = current->length - 1;
325  }
326  else {
327  --state;
328  }
329 
330  state_changed = 1;
331  joystick_centered = 0;
332  break;
333 
334  case DOWN:
335  if (!joystick_centered) {
336  break;
337  }
338 
339  if (state == current->length - 1) {
340  state = 0;
341  }
342  else {
343  ++state;
344  }
345 
346  state_changed = 1;
347  joystick_centered = 0;
348  break;
349 
350  case CENTER:
351  joystick_centered = 1;
352  break;
353 
354  default:
355  break;
356  }
357 
358  if (state_changed) {
359  if (is_in_highscore_menu()) {
360  highscore_menu_write_to_sram();
361  }
362  else {
363  menu_write_to_sram();
364  }
365  }
366 }
